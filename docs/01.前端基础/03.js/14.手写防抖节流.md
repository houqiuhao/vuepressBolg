> 话不多说，开撸
## 防抖
```js
function debounce(fn, time) {
	let that = this
	// 清除定时器
	clearTimeout(fn.timeout)
	// 获取参数
	let args = arguments
	fn.timeout = setTimeout(() => {
		fn.apply(that,args)
	},time)
}
```

## 节流
```js
function throttle(fn, delay) {
	let last,deferTimer
	return function () {
		let that = this
		// let now = new Date()  ❗❗这边是需要获取日期的值 +new Date()相当于获取值
		let now = +new Date()
		args = arguments
		if(last && now < last + delay) {
			clearTimeout(deferTimer)
			deferTimer = setTimeout(() => {
				fn.call(that, args)
				last = now
			},delay)
		} else {
			last = now
			fn.call(that, args)
		}
	}
}
```
::: tip
```js
	+ new Date()  //  +new Date() 会调用Date.prototype 上面的 valueOf方法
	// 相当于以下写法
	new Date().getTime()
	new Date().valueOf()
	new Date() * 1
```
