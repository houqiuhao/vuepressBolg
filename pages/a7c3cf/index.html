    # 宏任务和微任务

    js有两种任务的执行模式：同步和异步
    异步中又分为两种，微任务Microtask（和宏任务（Macrotask），宏任务是由宿主（浏览器，Node）发起的，而微任务是由js自身发起;宏任务会在下次事件循环开始的时候执行，微任务会在此次事件循环中同步任务结束后执行

    宏任务(Macrotas) 微任务(Microtask)
    setTimeout requestAnimationFrame（有争议）
    setInterval MutationObserver（浏览器环境）
    MessageChannel Promise.[ then/catch/finally ]
    I/O，事件队列 process.nextTick（Node环境）
    setImmediate（Node环境） queueMicrotask
    script（整体代码块）

    # 手撕Promise

    我们使用的 Promise 是基于 Promise A+ 规范实现的

    flow 掘金 IT老班长大佬的文章学着手写一版，大佬说道：很多手写版本都是使用 setTimeout 去做异步处理，但是 setTimeout 属于宏任务，这与 Promise 是个微任务相矛盾，所以打算选择一种创建微任务的方式去实现我们的手写代码。
    这里我们有几种选择，一种就是 Promise A+ 规范中也提到的，process.nextTick（ Node 端 ） 与MutationObserver（ 浏览器端 ），考虑到利用这两种方式需要做环境判断，所以在这里我们就推荐另外一种创建微任务的方式 queueMicrotask

    基本原理：
1. Promise 是一个类，在执行这个类的时候会传入一个执行器，这个执行器会立即执行
2. Promise 会有三种状态
    Pending 等待
    Fulfilled 完成
    Rejected 失败
3. 状态只能由 Pending --> Fulfilled 或者 Pending --> Rejected，且一但发生改变便不可二次修改；
4. Promise 中使用 resolve 和 reject 两个函数来更改状态；
5. then 方法内部做但事情就是状态判断
    如果状态是成功，调用成功回调函数
    如果状态是失败，调用失败回调函数

    # 一、核心简单实现

    const FULFILLED = 'fulfilled'
const PENDING = 'pending'
const REJECTED = 'rejected'
class MyPromise {
    //构造函数
    constructor(executor){
        //立即执行函数，executor是执行器
        executor(this.resolve,this.reject)
    }
    // 最开始的状态是PENDING
    status = PENDING

    // 成功后的值 初始为空
    value = null

    // 失败后的原因 初始为空
    reason = null

    // resolve和reject为什么要用箭头函数？
    // 如果直接调用的话，普通函数this指向的是window或者undefined
    // 用箭头函数就可以让this指向当前实例对象

    // 更改成功后的状态
    resolve = (value) => {
        //成功后状态会边为fulfilled
        // !只有状态为pending才能更改状态
        if(this.status === PENDING) {
            this.status = FULFILLED
            // 成功后返回值
            // ！保存成功的返回值
            this.value = value
        }
    }
    // 更改失败后的状态
    reject = (reason) =>{
        // 更新失败后的状态
        // 只能改变pending的状态
        if(this.status === PENDING) {
            this.status = REJECTED
            // 返回失败原因
            // !保存失败的原因
            this.reason = reason
        }
    }
    // 处理then 
    // ！THEN是个函数，且会处理onFulfilled和onRejected
    then(onFulfilled,onRejected) {
        // !需要判断状态
        // 如果状态是成功则调用onFulfilled
        if(this.status === FULFILLED) {
            // 调用函数
            onFulfilled(this.value)
        } else if (this.status === REJECTED) {
            onRejected(this.reason)
        }
    }


}
module.exports = MyPromise

    上面代码可以实现简单的新建promise了可以用如下代码测试：

    const promise = new MyPromise((resolve, reject) => {
   resolve('success')
   reject('err')
})

promise.then(value => {
  console.log('resolve', value)
}, reason => {
  console.log('reject', reason)
})
// 执行结果：resolve success

    # 三、then方法多次调用添加多个处理函数

    const FULFILLED = 'fulfilled'
const PENDING = 'pending'
const REJECTED = 'rejected'
class MyPromise {
    //构造函数
    constructor(executor){
        //立即执行函数，executor是执行器
        executor(this.resolve,this.reject)
    }
    // 最开始的状态是PENDING
    status = PENDING
    // 成功后的值 初始为空
    value = null
    // 失败后的原因 初始为空
    reason = null

    // then修改 需要为数组
    // 缓存成功回调函数
    onFulfilledCallback = []
    // 缓存失败回调函数
    onRejectedCallback = []


    // resolve和reject为什么要用箭头函数？
    // 如果直接调用的话，普通函数this指向的是window或者undefined
    // 用箭头函数就可以让this指向当前实例对象
    // 更改成功后的状态
    resolve = (value) => {
        //成功后状态会边为fulfilled
        // !只有状态为pending才能更改状态
        if(this.status === PENDING) {
            this.status = FULFILLED
            // 成功后返回值
            // ！保存成功的返回值
            this.value = value
            // 调用回调函数
            // 修改调用
            while(this.onFulfilledCallback.length){
                this.onFulfilledCallback.shift()(value)
                
            }
        }
    }
    // 更改失败后的状态
    reject = (reason) =>{
        // 更新失败后的状态
        // 只能改变pending的状态
        if(this.status === PENDING) {
            this.status = REJECTED
            // 返回失败原因
            // !保存失败的原因
            this.reason = reason
            // 调用回调函数
            // 修改调用
            while(this.onRejectedCallback.length) {
                this.onRejectedCallback.shift()(reason)

            }
        }
    }
    // 处理then 
    // ！THEN是个函数，且会处理onFulfilled和onRejected
    then(onFulfilled,onRejected) {
        // !需要判断状态
        // 如果状态是成功则调用onFulfilled
        if(this.status === FULFILLED) {
            // 调用函数
            onFulfilled(this.value)
        } else if (this.status === REJECTED) {
            onRejected(this.reason)
        } else if (this.status === PENDING) {
            // 添加pending时候的处理
            this.onFulfilledCallback.push(onFulfilled)
            this.onRejectedCallback.push(onFulfilled)
        }    
    }
}
module.exports = MyPromise

    # 四、then方法的链式调用 && 五、then链式调用识别Promise是否返回自己 && 六、错误捕获

    then 方法要链式调用那么就需要返回一个 Promise 对象
then 方法里面 return 一个返回值作为下一个 then 方法的参数，如果是 return 一个 Promise 对象，那么就需要判断它的状态
-------------------------------------------------------------------------------
如果 then 方法返回的是自己的 Promise 对象，则会发生循环调用，这个时候程序会报错

    const FULFILLED = 'fulfilled'
const PENDING = 'pending'
const REJECTED = 'rejected'
class MyPromise {
    //构造函数
    constructor(executor){
        //立即执行函数，executor是执行器
        try {
            executor(this.resolve,this.reject)
        } catch (error) {
            this.reject(error)
        }
    }
    // 最开始的状态是PENDING
    status = PENDING
    // 成功后的值 初始为空
    value = null
    // 失败后的原因 初始为空
    reason = null

    // then修改 需要为数组
    // 缓存成功回调函数
    onFulfilledCallback = []
    // 缓存失败回调函数
    onRejectedCallback = []


    // resolve和reject为什么要用箭头函数？
    // 如果直接调用的话，普通函数this指向的是window或者undefined
    // 用箭头函数就可以让this指向当前实例对象
    // 更改成功后的状态
    resolve = (value) => {
        //成功后状态会边为fulfilled
        // !只有状态为pending才能更改状态
        if(this.status === PENDING) {
            this.status = FULFILLED
            // 成功后返回值
            // ！保存成功的返回值
            this.value = value
            // 调用回调函数
            // 修改调用
            while(this.onFulfilledCallback.length){
                this.onFulfilledCallback.shift()(value)
                
            }
        }
    }
    // 更改失败后的状态
    reject = (reason) =>{
        // 更新失败后的状态
        // 只能改变pending的状态
        if(this.status === PENDING) {
            this.status = REJECTED
            // 返回失败原因
            // !保存失败的原因
            this.reason = reason
            // 调用回调函数
            // 修改调用
            while(this.onRejectedCallback.length) {
                this.onRejectedCallback.shift()(reason)

            }
        }
    }
    // 处理then 
    // ！THEN是个函数，且会处理onFulfilled和onRejected
    // ===新增===
    // 链式调用就得返回promise 判断状态进行处理
    then(onFulfilled,onRejected) {
        // !需要判断状态
        // === 新增 ===
        // 先创建promise
        const promise2 = new MyPromise((resolve,reject) => {
            // 如果状态是成功则调用onFulfilled
            if(this.status === FULFILLED) {
                // 调用函数
                // 保存返回的处理值

                // 五、需要创建微任务等待promise2创建完毕
                queueMicrotask(() => {
                    try {
                        const x = onFulfilled(this.value)
                        // 用一个函数处理值，需要判断是promise还是普通函数
                        resolvePromise(promise2, x, resolve, reject)
                    } catch(error) {
                        //! 调用的传入的reject 而不是this.reject
                        reject(error)
                    }
                })
            } else if (this.status === REJECTED) {
                onRejected(this.reason)
            } else if (this.status === PENDING) {
                // 添加pending时候的处理
                this.onFulfilledCallback.push(onFulfilled)
                this.onRejectedCallback.push(onFulfilled)
            }
        })
        // ！！要将promise2 return出来
        return promise2
    }
}
// === 新增 ===
// 创建函数处理promise
function resolvePromise (promise2, x, resolve, reject) {
    // 如果相等了，说明return的是自己，抛出类型错误并返回
    if (promise2 === x) {
        return reject(new TypeError('Chaining cycle detected for promise #<Promise>'))
    }
    // 判断是否promise
    if(x instanceof MyPromise) {
        // 执行 x，调用 then 方法，目的是将其状态变为 fulfilled 或者 rejected
        // x.then(value => resolve(value), reason => reject(reason))
        // 简化之后
        x.then(resolve, reject)
    } else {
        // 普通值
        resolve(x)
    }
}
module.exports = MyPromise

    # 七、考虑fufilled状态下的处理，对rejected 和pending改造

    增加异步状态下的链式调用
增加回调函数执行结果的判断
增加识别 Promise 是否返回自己
增加错误捕获

    const FULFILLED = 'fulfilled'
const PENDING = 'pending'
const REJECTED = 'rejected'
class MyPromise {
    //构造函数
    constructor(executor){
        //立即执行函数，executor是执行器
        try {
            executor(this.resolve,this.reject)
        } catch (error) {
            this.reject(error)
        }
    }
    // 最开始的状态是PENDING
    status = PENDING
    // 成功后的值 初始为空
    value = null
    // 失败后的原因 初始为空
    reason = null

    // then修改 需要为数组
    // 缓存成功回调函数
    onFulfilledCallback = []
    // 缓存失败回调函数
    onRejectedCallback = []


    // resolve和reject为什么要用箭头函数？
    // 如果直接调用的话，普通函数this指向的是window或者undefined
    // 用箭头函数就可以让this指向当前实例对象
    // 更改成功后的状态
    resolve = (value) => {
        //成功后状态会边为fulfilled
        // !只有状态为pending才能更改状态
        if(this.status === PENDING) {
            this.status = FULFILLED
            // 成功后返回值
            // ！保存成功的返回值
            this.value = value
            // 调用回调函数
            // 修改调用
            while(this.onFulfilledCallback.length){
                this.onFulfilledCallback.shift()(value)
                
            }
        }
    }
    // 更改失败后的状态
    reject = (reason) =>{
        // 更新失败后的状态
        // 只能改变pending的状态
        if(this.status === PENDING) {
            this.status = REJECTED
            // 返回失败原因
            // !保存失败的原因
            this.reason = reason
            // 调用回调函数
            // 修改调用
            while(this.onRejectedCallback.length) {
                this.onRejectedCallback.shift()(reason)

            }
        }
    }
    // 处理then 
    // ！THEN是个函数，且会处理onFulfilled和onRejected
    // ===新增===
    // 链式调用就得返回promise 判断状态进行处理
    then(onFulfilled,onRejected) {
        // !需要判断状态
        // === 新增 ===
        // 先创建promise
        const promise2 = new MyPromise((resolve,reject) => {
            // 如果状态是成功则调用onFulfilled
            if(this.status === FULFILLED) {
                // 调用函数
                // 保存返回的处理值

                // 五、需要创建微任务等待promise2创建完毕
                queueMicrotask(() => {
                    try {
                        const x = onFulfilled(this.value)
                        // 用一个函数处理值，需要判断是promise还是普通函数
                        resolvePromise(promise2, x, resolve, reject)
                    } catch(error) {
                        //! 调用的传入的reject 而不是this.reject
                        reject(error)
                    }
                })
            } else if (this.status === REJECTED) {
                // 同上面 增加微任务
                queueMicrotask(() => {
                    try {
                        const x = onRejected(this.reason)
                        // 用一个函数处理值，需要判断是promise还是普通函数
                        resolvePromise(promise2, x, resolve, reject)
                    } catch(error) {
                        //! 调用的传入的reject 而不是this.reject
                        reject(error)
                    }
                })
            } else if (this.status === PENDING) {
                // 添加pending时候的处理
                // === 新增 ===
                // ===需要push进去的应该是微任务===
                this.onFulfilledCallback.push(() => {
                    queueMicrotask(() => {
                        try {
                            // 获取成功回调函数的执行结果
                            const x = onFulfilled(this.value);
                            // 传入 resolvePromise 集中处理
                            resolvePromise(promise2, x, resolve, reject);
                        } catch (error) {
                            reject(error)
                        }
                    })
                })
                this.onRejectedCallback.push(() => {
                    queueMicrotask(() => {
                        try {
                            // 调用失败回调，并且把原因返回
                            const x = onRejected(this.reason);
                            // 传入 resolvePromise 集中处理
                            resolvePromise(promise2, x, resolve, reject);
                        } catch (error) {
                            reject(error)
                        }
                    })
                })
            }
        })
        // ！！要将promise2 return出来
        return promise2
    }
}
// 创建函数处理promise
function resolvePromise (promise2, x, resolve, reject) {
    // 如果相等了，说明return的是自己，抛出类型错误并返回
    if (promise2 === x) {
        return reject(new TypeError('Chaining cycle detected for promise #<Promise>'))
    }
    // 判断是否promise
    if(x instanceof MyPromise) {
        // 执行 x，调用 then 方法，目的是将其状态变为 fulfilled 或者 rejected
        // x.then(value => resolve(value), reason => reject(reason))
        // 简化之后
        x.then(resolve, reject)
    } else {
        // 普通值
        resolve(x)
    }
}
module.exports = MyPromise

    // then修改 需要为数组
    // 缓存成功回调函数
    onFulfilledCallback = []
    // 缓存失败回调函数
    onRejectedCallback = []


    // resolve和reject为什么要用箭头函数？
    // 如果直接调用的话，普通函数this指向的是window或者undefined
    // 用箭头函数就可以让this指向当前实例对象
    // 更改成功后的状态
    resolve = (value) => {
        //成功后状态会边为fulfilled
        // !只有状态为pending才能更改状态
        if(this.status === PENDING) {
            this.status = FULFILLED
            // 成功后返回值
            // ！保存成功的返回值
            this.value = value
            // 调用回调函数
            // 修改调用
            while(this.onFulfilledCallback.length){
                this.onFulfilledCallback.shift()(value)
                
            }
        }
    }
    // 更改失败后的状态
    reject = (reason) =>{
        // 更新失败后的状态
        // 只能改变pending的状态
        if(this.status === PENDING) {
            this.status = REJECTED
            // 返回失败原因
            // !保存失败的原因
            this.reason = reason
            // 调用回调函数
            // 修改调用
            while(this.onRejectedCallback.length) {
                this.onRejectedCallback.shift()(reason)

            }
        }
    }
    // 处理then 
    // ！THEN是个函数，且会处理onFulfilled和onRejected
    // ===新增===
    // 链式调用就得返回promise 判断状态进行处理
    then(onFulfilled,onRejected) {
        // !需要判断状态
        // 八 === 新增 ===
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value;
        onRejected = typeof onRejected === 'function' ? onRejected : reason => {throw reason};
        // 先创建promise
        const promise2 = new MyPromise((resolve,reject) => {
            // 如果状态是成功则调用onFulfilled
            if(this.status === FULFILLED) {
                // 调用函数
                // 保存返回的处理值

                // 五、需要创建微任务等待promise2创建完毕
                queueMicrotask(() => {
                    try {
                        const x = onFulfilled(this.value)
                        // 用一个函数处理值，需要判断是promise还是普通函数
                        resolvePromise(promise2, x, resolve, reject)
                    } catch(error) {
                        //! 调用的传入的reject 而不是this.reject
                        reject(error)
                    }
                })
            } else if (this.status === REJECTED) {
                // 同上面 增加微任务
                queueMicrotask(() => {
                    try {
                        const x = onRejected(this.reason)
                        // 用一个函数处理值，需要判断是promise还是普通函数
                        resolvePromise(promise2, x, resolve, reject)
                    } catch(error) {
                        //! 调用的传入的reject 而不是this.reject
                        reject(error)
                    }
                })
            } else if (this.status === PENDING) {
                // 添加pending时候的处理
                // === 新增 ===
                // ===需要push进去的应该是微任务===
                this.onFulfilledCallback.push(() => {
                    queueMicrotask(() => {
                        try {
                            // 获取成功回调函数的执行结果
                            const x = onFulfilled(this.value);
                            // 传入 resolvePromise 集中处理
                            resolvePromise(promise2, x, resolve, reject);
                        } catch (error) {
                            reject(error)
                        }
                    })
                })
                this.onRejectedCallback.push(() => {
                    queueMicrotask(() => {
                        try {
                            // 调用失败回调，并且把原因返回
                            const x = onRejected(this.reason);
                            // 传入 resolvePromise 集中处理
                            resolvePromise(promise2, x, resolve, reject);
                        } catch (error) {
                            reject(error)
                        }
                    })
                })
            }
        })
        // ！！要将promise2 return出来
        return promise2
    }
    // 九、增加静态方法
    static resolve (parameter) {
        if(parameter instanceof MyPromise) {
            return parameter
        }
        return new MyPromise(resovle => {
            resolve(parameter)
        })
    }
    static reject (reason) {
        return new MyPromise((resolve, reject) => {
            reject(reason)
        })
    }
}
// 创建函数处理promise
function resolvePromise (promise2, x, resolve, reject) {
    // 如果相等了，说明return的是自己，抛出类型错误并返回
    if (promise2 === x) {
        return reject(new TypeError('Chaining cycle detected for promise #<Promise>'))
    }
    // // 判断是否promise
    // if(x instanceof MyPromise) {
    //     // 执行 x，调用 then 方法，目的是将其状态变为 fulfilled 或者 rejected
    //     // x.then(value => resolve(value), reason => reject(reason))
    //     // 简化之后
    //     x.then(resolve, reject)

    // ========这波居然要改成 ===========
    if (typeof x === 'object' || typeof x === 'function') {
        // x 为 null 直接返回，走后面的逻辑会报错
        if (x === null) {
            return resolve(x);
        }
        let then;
        try {
            // 把 x.then 赋值给 then 
            then = x.then;
        } catch (error) {
            // 如果取 x.then 的值时抛出错误 error ，则以 error 为据因拒绝 promise
            return reject(error);
        }
        // 如果 then 是函数
        if (typeof then === 'function') {
            let called = false;
            try {
                then.call(
                x, // this 指向 x
                // 如果 resolvePromise 以值 y 为参数被调用，则运行 [[Resolve]](promise, y)
                y => {
                    // 如果 resolvePromise 和 rejectPromise 均被调用，
                    // 或者被同一参数调用了多次，则优先采用首次调用并忽略剩下的调用
                    // 实现这条需要前面加一个变量 called
                    if (called) return;
                    called = true;
                    resolvePromise(promise, y, resolve, reject);
                },
                // 如果 rejectPromise 以据因 r 为参数被调用，则以据因 r 拒绝 promise
                r => {
                    if (called) return;
                    called = true;
                    reject(r);
                });
            } catch (error) {
                // 如果调用 then 方法抛出了异常 error：
                // 如果 resolvePromise 或 rejectPromise 已经被调用，直接返回
                if (called) return;
                // 否则以 error 为据因拒绝 promise
                reject(error);
            }
        } else {
            // 如果 then 不是函数，以 x 为参数执行 promise
            resolve(x);
        }
    } else {
        // 普通值
        resolve(x)
    }
}
